1. 什么是CoffeeScript？

CoffeeScript是一个使用纯Ruby编写的新编程语言，创建者Jeremy Ashkenas戏称它是JavaScript的不那么卖弄的小兄弟。

CoffeeScript会将编写的代码编译成JavaScript，也就是说将JavaScript当成一种中间语言。

CoffeeScript用一种简单的方式来呈现JavaScript的Good Parts，编译后的代码可以无警告地通过JavaScript Lint的检查。

注意：最初的编译器确实是由Ruby所写，但是v0.5.0之后，都是用CoffeeScript自身来编写。实际上，利用旧版本的编译器，编译出一个拥有更多特性，新版本的编译器，这个过程叫Bootstrapping。

  return x * x;
};

cube = function(x) {
  return square(x) * x;
};





var fill;

fill = function(container, liquid) {
  if (liquid == null) liquid = "coffee";
  return "Filling the " + container + " with " 
         + liquid + "...";
};

注意：左边为CoffeeScript，右边为编译后的JavaScript。

class Shape
  constructor: (@width) ->
  computeArea: -> 
    throw new Error('I am an abstract class!')

class Square extends Shape
  computeArea: -> Math.pow @width, 2
 
class Circle extends Shape
  radius: -> @width / 2
  computeArea: -> 
    Math.PI * Math.pow @radius(), 2
  showArea = (shape) ->
    unless shape instanceof Shape
      throw new Error('showArea requires a Shape instance!')
    console.log shape.computeArea()

showArea new Square(2) # 4
showArea new Circle(2) # pi

2.特性与优点—只保留JS中Good Parts

Account = (customer, cart) ->
  @customer = customer
  @cart = cart

  $('.shopping_cart').bind 'click', (event) =>
    @customer.purchase @cart


var Account;

Account = function(customer, cart) {
  var _this = this;
  this.customer = customer;
  this.cart = cart;

  return $('.shopping_cart').bind('click', 
    function(event) {
      return _this.customer.purchase(_this.cart);
  });
};

在JavaScript中，this关键字会动态地指向当前函数所属的对象。如果你把一个函数作为回调函数传给不同的对象，那么原来this所指向的值将会丢失。

CoffeeScript提供了一个"=>"来解决这个问题，如果函数通过"=>"来创建，那么这个函数则能直接访问函数定义时的this的属性，不会随着不同的所属对象造成this指向的不同。
